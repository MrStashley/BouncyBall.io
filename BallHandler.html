var gravity = .5; var clickX = 0; var clickY = 0; var r,g,b; var ballHolder = []; var newBall = false; canvas.onmousedown = function(e){ if(e.button == 0){ clickX = e.clientX; clickY = e.clientY; newBall = true; } }; ballHolder[0] = new ball(1000,100,0,4,0,gravity,"rgb("+Math.floor((Math.random() * 255) + 1)+","+Math.floor((Math.random() * 255) + 1)+","+Math.floor((Math.random() * 255) + 1)+")"); function createBall(){ r = Math.floor((Math.random() * 255) + 1); g = Math.floor((Math.random() * 255) + 1); b = Math.floor((Math.random() * 255) + 1); if(newBall){ ballHolder[ballHolder.length] = new ball(clickX,clickY,0,0,0,gravity,"rgb("+r+","+g+","+b+")"); newBall = false; } } setInterval(createBall,30); function draw(){ c.beginPath(); c.fillStyle = "#eee"; c.fillRect(0,0,canvas.width,canvas.height); c.closePath(); for (var i = 0; i < ballHolder.length; i++) { ballHolder[i].draw(); ballHolder[i].move(); } requestAnimationFrame(draw); } draw();
